\chapter{Casos de pruebas automatizadas}

En este capitulo se presenta la especificación formal de los casos de prueba automatizados realizados.

\begin{prueba}
    \TestID{\tcounter}
    \Titulo{Validación del parametro del metodo \textit{setProperty} de la clase \textit{Pump}.}
    \Caracteristica{Validación del parametro enviado como valor al método \textit{setProperty} cuando se usa la clave \textit{PumpProperty.HEAD}.}
    \Objetivo{Comprobar que el parametro enviado como valor tiene el recibe el tipo correcto cuando se usa la clave \textit{PumpProperty.HEAD}.}
    \Configuracion{
        Clave: \textit{PumpProperty.Head}\\
        Value: Un objeto diferente a un String\\
    }
    \DatosPrueba{}
    \AccionesPrueba{}
    \ResultadosEsperados{}
\end{prueba}

\begin{prueba}
    \TestID{\tcounter}
    \Titulo{Correct Login}
    \Caracteristica{Login to umkc.manageMyID.com with Roo Balance App}
    \Objetivo{Confirm that proper user id an password yields access to the website as expected.}
    \Configuracion{IOS device or simulator has Roo Balance application ready to run.}
    \DatosPrueba{Login information\\
        User Email = cdbpc@umkc.edu\\
        Password = r00bucks
    }
    \AccionesPrueba{1. Start Roo Balance application\\
        2. Select View Account option\\
        3. Enter login information
    }
    \ResultadosEsperados{System displays account balance with option to logoff.  Account balance should match that found on umkc.ManageMyID.com.}
\end{prueba}
